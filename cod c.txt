#include <reg51.h>
#include <stdio.h>

// PINII LCD
#define LCD_PORT P1
sbit LCD_E = P1^3;
sbit LCD_RS = P1^2;

// VARIABILE
unsigned char * singleChar;
signed int temp;
char display_buffer[33];
float currentTemp, setTemp;
float histerezis = 0.5f; 

// SENZOR ONE-WIRE (DS18B20)
unsigned char sensorBuffer[2];
sbit pin_1W=P3^3;

// RELEU
sbit pin_relay = P3^7;

// LED-URI
sbit led_albastru = P3^4; 
sbit led_rosu     = P3^5; // Încalzire / Releu ON

// TASTATURA 
sbit key_dec = P1^0;
sbit key_inc = P1^1;

// FUNCTII
void delay(unsigned int i);
void init_1W(void);
unsigned char readByte_1W(void);
void writeByte_1W(unsigned char dat);
void read_sensor(void);
void lcd_cmd(unsigned char cmd);
void lcd_chars(unsigned char * chars);
void lcd_init();
void get_keys();
void controleaza_sistem();

void main()
{
	// INITIALIZARE STARI INITIALE
	pin_relay = 0;
	led_albastru = 0;
	led_rosu = 0;
	
	LCD_PORT = 0x0F;
	LCD_E = 0;
	LCD_RS = 0;
	
	key_dec = 1;
	key_inc = 1;
	setTemp = 24.0f; // Temperatura tinta initiala
	lcd_init();

	while(1)
	{
		// Citire senzor si conversie temperatura
		read_sensor();
		temp = sensorBuffer[0]; // lsb
		temp &= 0x00FF;
		temp |= (sensorBuffer[1] << 8);
		currentTemp = temp * 0.0625;
		
		// Afisare Linie 1 LCD (Temperatura curenta)
		sprintf(display_buffer, "Temperatura = %.2f", currentTemp);
		lcd_cmd(0x01); // Clear display
		lcd_cmd(0x80); // Prima linie
		lcd_chars(display_buffer);
		
		// Afisare Linie 2 LCD (Temperatura setata)
		lcd_cmd(0xC0); // A doua linie
		sprintf(display_buffer, "Set Temp = %.2f", setTemp);
		lcd_chars(display_buffer);
		
		// Control releu si leduri pe baza histerezisului
		controleaza_sistem();
		
		// Citire ?i debouncing butoane
		get_keys();
		
		// Mica pauza pentru fluiditatea simularii în Proteus
		delay(5000);
	}
}

// --- LOGICA CONTROL RELEU SI LED-URI CU HISTEREZIS ---
void controleaza_sistem() 
{
	if (currentTemp < (setTemp - histerezis)) 
	{
		pin_relay = 1;     // Pornim caldura
		led_rosu = 1;      // LED Ro?u aprins
		led_albastru = 0;  // LED Albastru stins
	} 
	else if (currentTemp >= (setTemp + histerezis)) 
	{
		pin_relay = 0;     // Oprim caldura
		led_rosu = 0;      // LED Ro?u stins
		led_albastru = 1;  // LED Albastru aprins
	}
}

// FUNCTIE IMPLEMENTARE
void delay(unsigned int i)
{
	while(i--);
}

// INITIALIZARE ONE-WIRE
void init_1W(void)
{
	pin_1W = 1;
	delay(8);
	pin_1W = 0;
	delay(80);
	pin_1W = 1;
	delay(14);
	delay(20);
}

// CITIRE OCTET ONE-WIRE
unsigned char readByte_1W(void)
{
	unsigned char i=0;
	unsigned char dat = 0;
	for (i=8;i>0;i--)
	{
		pin_1W = 0;
		dat>>=1;
		pin_1W = 1;
		if(pin_1W)
			dat|=0x80;
		delay(4);
	}
	return(dat);
}

// SCRIERE OCTET ONE-WIRE
void writeByte_1W(unsigned char dat)
{
	unsigned char i=0;
	for (i=8; i>0; i--)
	{
		pin_1W = 0;
		pin_1W = dat&0x01;
		delay(5);
		pin_1W = 1;
		dat>>=1;
	}
	delay(4);
}

// CITIRE SENZOR DS18B20
void read_sensor(void)
{
	init_1W();
	writeByte_1W(0xCC); // Skip ROM
	writeByte_1W(0x44); // Convert T
	
	// Timp de asteptare pentru ca senzorul sa finalizeze conversia în Proteus
	delay(2000); 
	
	init_1W();
	writeByte_1W(0xCC); // Skip ROM
	writeByte_1W(0xBE); // Read Scratchpad
	sensorBuffer[0]=readByte_1W();
	sensorBuffer[1]=readByte_1W();
}

// FUNCTII COMANDA LCD
void lcd_cmd(unsigned char cmd)
{
	LCD_RS = 0;
	LCD_PORT &= 0x0F;
	LCD_PORT |= (cmd & 0xF0);
	LCD_E = 1;
	delay(100);
	LCD_E = 0;
	delay(1000);
	LCD_PORT &= 0x0F;
	LCD_PORT |= ((cmd << 4) & 0xF0);
	LCD_E = 1;
	delay(100);
	LCD_E = 0;
	delay(1000);
}

// TRIMITERE TEXT PE LCD
void lcd_chars(unsigned char * chars)
{
	LCD_RS = 1;
	singleChar = chars;
	while(*singleChar != '\0')
	{
		LCD_PORT &= 0x0F;
		LCD_PORT |= ((*singleChar) & 0xF0);
		LCD_RS = 1;
		LCD_E = 1;
		delay(100);
		LCD_E = 0;
		delay(1000);
		LCD_PORT &= 0x0F;
		LCD_PORT |= (((*singleChar) << 4) & 0xF0);
		LCD_RS = 1;
		LCD_E = 1;
		delay(100);
		LCD_E = 0;
		delay(1000);
		singleChar++;
	}
}

// INITIALIZARE COMPATIBILA LM016L
void lcd_init()
{
	lcd_cmd(0x02); // Mod 4 biTi
	lcd_cmd(0x28); // Caractere 5x7
	lcd_cmd(0x0E); // Cursor activat
	lcd_cmd(0x01); // CuraTa ecranul
	lcd_cmd(0x80); // Prima poziTie
}

// CITIRE SI DEBOUNCING BUTOANE (LIMITA 5°C - 40°C)
void get_keys()
{
	key_inc = 1;
	key_dec = 1;
	
	if(!key_inc)
	{
		delay(1000);
		if (!key_inc)
		{
			setTemp += 0.5f;
			if (setTemp > 40.0f) 
			{
				setTemp = 40.0f; // Limita maxima
			}
		}
		return;
	}
	
	if(!key_dec)
	{
		delay(1000);
		if (!key_dec)
		{
			setTemp -= 0.5f;
			if (setTemp < 5.0f) 
			{
				setTemp = 5.0f; // Limita minima
			}
		}
		return;
	}
}
