; LCD pins
LCD_PORT equ P1
LCD_E equ P1.3
LCD_RS equ P1.2

; OneWire pin
PIN_1W equ  P3.3

; Heater relay pin
PIN_RELAY equ P3.7

; LED pins
LED_ALBASTRU equ P3.4
LED_ROSU     equ P3.5

; Keyboard pins
KEY_DEC equ P1.0
KEY_INC equ P1.1

; Timer consts for generic 8051 for 1 ms delay
INIT_TL0 equ 017h
INIT_TH0 equ 0FCh

; Sensor data buffer start address
SENSOR_DATA_BUFFER equ 030h

; Set temperature buffer start address
SET_TEMP_BUFFER equ 032h

; Display buffer start address
DISP_BUFFER_L1 equ 034h
DISP_BUFFER_L2 equ 045h

org 0000h
sjmp main

text1:
db "Temperatura = ", 0
text2:
db "Set Temp = ", 0

main:
    ; I/O pins initialization
    mov LCD_PORT, #00Fh ; Seteaza pinii de date pe high conform codului C
    clr LCD_E
    clr LCD_RS
    clr PIN_RELAY
    clr LED_ALBASTRU
    clr LED_ROSU
    setb KEY_DEC
    setb KEY_INC

    ; Set temperature buffer initialization (24 grade)
    ; 24 grade / 0.0625 = 384 = 0180h
    mov r0, #SET_TEMP_BUFFER
    mov @r0, #080h       ; LSB
    inc r0
    mov @r0, #001h       ; MSB

    ; Timer initialization
    orl TMOD, #00010001b
    anl TMOD, #11111101b
    mov P0, #0

    ; Display initialization
    acall lcd_init

main_loop:
    acall read_sensor
    ; copy sensor data to buffer
    mov r0, #SENSOR_DATA_BUFFER
    mov a, r1
    mov @r0, a
    mov a, r2
    inc r0
    mov @r0, a

    ; clear LCD
    mov a, #01h
    clr LCD_RS
    acall lcd_send_byte

    ; Copy constant text for line 1
    mov r0, #DISP_BUFFER_L1
    mov dptr, #text1
    setb LCD_RS
    acall lcd_copy_buffer

    ; Convert sensor data to ASCII
    acall convert_int_temp
    acall to_ascii
    acall load_int_disp_buffer

    acall convert_frac_temp
    acall to_ascii
    acall load_frac_disp_buffer
    mov @r0, #0

    ; Display line 1
    mov r0, #DISP_BUFFER_L1
    acall lcd_send_chars_ram 

    ; Load set temperature to registers for conversion
    mov r0, #SET_TEMP_BUFFER
    mov a, @r0
    mov r1, a
    inc r0
    mov a, @r0
    mov r2, a

    ; Move to line 2
    mov a, #0C0h
    clr LCD_RS
    acall lcd_send_byte

    ; Copy constant text for line 2
    mov r0, #DISP_BUFFER_L2
    mov dptr, #text2
    acall lcd_copy_buffer

    ; Convert set temperature to ASCII
    acall convert_int_temp
    acall to_ascii
    acall load_int_disp_buffer

    acall convert_frac_temp
    acall to_ascii
    acall load_frac_disp_buffer
    mov @r0, #0

    ; Display line 2
    mov r0, #DISP_BUFFER_L2
    acall lcd_send_chars_ram

    ; Control relay and LEDs based on hysteresis
    acall controleaza_sistem
    
    ; Read keyboard
    acall get_keys

    ; Wait ~100 ms 
    mov r3, #100
    delay_loop:
        acall delay_1ms
        djnz r3, delay_loop
    ljmp main_loop


delay_1ms:
    mov TL0, #INIT_TL0
    mov TH0, #INIT_TH0
    setb TR0
    wait:
        jnb TF0, wait
    clr TR0
    clr TF0
    ret

simple_delay:
    djnz r7, simple_delay
    ret

init_1W:
    setb PIN_1W
    mov r7, #48
    acall simple_delay
    clr PIN_1W
    mov r7, #180
    acall simple_delay
    mov r7, #180
    acall simple_delay
    setb PIN_1W
    mov r7, #84
    acall simple_delay
    ret

read_byte_1W:
    mov r6, #8
    clr a
    start_read:
        clr PIN_1W
        clr C
        rrc a
        setb PIN_1W
        jnb PIN_1W, read_end
        orl a, #080h
    read_end:
        mov r7, #30
        acall simple_delay
        djnz r6, start_read
    ret

write_byte_1W:
    mov r6, #8
    start_write:
        clr PIN_1W
        rrc a
        mov PIN_1W, C
        mov r7, #45
        acall simple_delay
        setb PIN_1W
        djnz r6, start_write
    mov r7, #30
    acall simple_delay
    ret

read_sensor:
    acall init_1W
    mov a, #0CCh ; skip ROM
    acall write_byte_1W
    mov a, #044h ; convert T
    acall write_byte_1W
    
    ; Delay pentru ca senzorul sa finalizeze conversia (echivalent delay(2000))
    mov r4, #50
    wait_conv:
        acall delay_1ms
        djnz r4, wait_conv

    acall init_1W
    mov a, #0CCh ; skip ROM
    acall write_byte_1W
    mov a, #0BEh ; read Scratchpad
    acall write_byte_1W
    acall read_byte_1W
    mov r1, a    ; LSB
    acall read_byte_1W
    mov r2, a    ; MSB
    ret

lcd_send_byte:
    anl LCD_PORT, #0Fh
    mov r7, a
    anl a, #0F0h
    mov b, LCD_PORT
    anl b, #0Fh
    orl a, b
    mov LCD_PORT, a
    mov a, r7
    setb LCD_E
    acall delay_1ms
    clr LCD_E

    mov r7, #4
    lcd_cmd_wait:
        acall delay_1ms
        djnz r7, lcd_cmd_wait

    anl LCD_PORT, #0Fh
    mov r7, #4
    lcd_shift:
        clr c
        rlc a
        djnz r7, lcd_shift
    anl a, #0F0h
    mov b, LCD_PORT
    anl b, #0Fh
    orl a, b
    mov LCD_PORT, a
    setb LCD_E
    acall delay_1ms
    clr LCD_E
    mov r7, #4
    lcd_cmd_wait2:
        acall delay_1ms
        djnz r7, lcd_cmd_wait2
    ret

lcd_init:
    clr LCD_RS
    mov a, #02h  ; 4 bit mode
    acall lcd_send_byte
    mov a, #028h ; 5x7 chars
    acall lcd_send_byte
    mov a, #0Eh  ; cursor on
    acall lcd_send_byte
    mov a, #01h  ; clear
    acall lcd_send_byte
    mov a, #080h ; move cursor to first position
    acall lcd_send_byte
    ret

lcd_copy_buffer:
    lcd_copy_loop:
        clr a
        movc a, @a+dptr
        jz lcd_copy_stop
        mov @r0, a
        inc r0
        inc dptr
        sjmp lcd_copy_loop
    lcd_copy_stop:
    ret

lcd_send_chars_ram:
    setb LCD_RS
    lcd_send_chars_ram_loop:
        mov a, @r0
        jz lcd_send_ram_stop
        acall lcd_send_byte
        inc r0
        sjmp lcd_send_chars_ram_loop
    lcd_send_ram_stop:
    ret

convert_int_temp:
    mov a, r2
    anl a, #080h
    clr PSW^1
    jnz conv_int_neg_temp
    sjmp conv_int_pos_temp
conv_int_neg_temp:
    setb PSW^1 
    mov a, r1
    cpl a
    clr c
    inc a
    mov r1, a
    mov a, r2
    cpl a
    addc a, #0
    mov r2, a
conv_int_pos_temp:
    mov a, r2
    mov r3, #4
    convert_int_temp_loop:
        clr c
        rlc a
        djnz r3, convert_int_temp_loop
    mov b, a
    mov a, r1
    mov r3, #4
    convert_int_temp_loop2:
        clr c
        rrc a
        djnz r3, convert_int_temp_loop2
    orl a, b
    ret

convert_frac_temp:
    mov a, r1
    anl a, #0Fh
    mov b, #6
    mul ab
    ret

to_ascii:
    mov b, #10
    div ab
    add a, #'0'
    mov r7, a
    mov a, b
    add a, #'0'
    mov b, a
    mov a, r7
    ret

load_int_disp_buffer:
    jb PSW^1, disp_minus
    sjmp skip
disp_minus:
    mov r3, a
    mov a, #'-'
    mov @r0, a
    inc r0
    mov a, r3
skip:
    mov @r0, a
    inc r0
    mov @r0, b
    inc r0
    ret

load_frac_disp_buffer:
    mov @r0, #'.'
    inc r0
    mov @r0, a
    inc r0
    mov @r0, b
    inc r0
    ret

; --- CITIRE TASTATURA CU LIMITARI (5C - 40C) ---
get_keys:
    setb KEY_INC
    setb KEY_DEC
    jb KEY_INC, check_key_dec
    
    ; Tasta INC apasata -> Debounce
    mov r3, #10
delay_inc:
    acall delay_1ms
    djnz r3, delay_inc
    jb KEY_INC, get_key_end

    ; Creste temperatura tinta
    mov r0, #SET_TEMP_BUFFER
    mov a, @r0
    add a, #8
    mov r4, a
    inc r0
    mov a, @r0
    addc a, #0
    mov r5, a

    ; Verificare limita maxima: 40 grade (0280h)
    clr c
    mov a, r5
    subb a, #02h
    jc save_inc
    jnz limit_max 
    mov a, r4
    subb a, #080h
    jc save_inc

limit_max:
    mov r4, #080h
    mov r5, #002h

save_inc:
    mov r0, #SET_TEMP_BUFFER
    mov a, r4
    mov @r0, a
    inc r0
    mov a, r5
    mov @r0, a
    ret

check_key_dec:
    jb KEY_DEC, get_key_end
    
    ; Tasta DEC apasata -> Debounce
    mov r3, #10
delay_dec:
    acall delay_1ms
    djnz r3, delay_dec
    jb KEY_DEC, get_key_end

    ; Scade temperatura tinta
    mov r0, #SET_TEMP_BUFFER
    mov a, @r0
    subb a, #8
    mov r4, a
    inc r0
    mov a, @r0
    subb a, #0
    mov r5, a

    ; Verificare limita minima: 5 grade (0050h)
    mov a, r5
    jnz save_dec 
    clr c
    mov a, r4
    subb a, #050h
    jnc save_dec

limit_min:
    mov r4, #050h
    mov r5, #000h

save_dec:
    mov r0, #SET_TEMP_BUFFER
    mov a, r4
    mov @r0, a
    inc r0
    mov a, r5
    mov @r0, a

get_key_end:
    ret

; --- LOGICA CONTROL CU HISTEREZIS ---
controleaza_sistem:
    ; --- TEST 1: currentTemp < (setTemp - histerezis) ---
    mov r0, #SET_TEMP_BUFFER
    mov a, @r0
    subb a, #8
    mov r4, a
    inc r0
    mov a, @r0
    subb a, #0
    mov r5, a

    ; Comparam currentTemp cu (setTemp - 8)
    mov r0, #SENSOR_DATA_BUFFER
    mov a, r4
    subb a, @r0
    mov r3, a
    inc r0
    mov a, r5
    subb a, @r0
    
    ; Adresa lui R5 in RAM direct este 05h (folosit in loc de operand r5 ilegal)
    jb ACC.7, test_histerezis_sus 
    mov a, r3
    orl a, 05h                    
    jz test_histerezis_sus        
    
    ; Porneste incalzirea
    setb PIN_RELAY
    setb LED_ROSU
    clr LED_ALBASTRU
    ret

test_histerezis_sus:
    ; --- TEST 2: currentTemp >= (setTemp + histerezis) ---
    mov r0, #SET_TEMP_BUFFER
    mov a, @r0
    add a, #8
    mov r4, a
    inc r0
    mov a, @r0
    addc a, #0
    mov r5, a

    ; Comparam currentTemp cu (setTemp + 8)
    mov r0, #SENSOR_DATA_BUFFER
    mov a, @r0
    subb a, r4
    inc r0
    mov a, @r0
    subb a, r5

    jb ACC.7, system_no_change
    
    ; Opreste incalzirea
    clr PIN_RELAY
    clr LED_ROSU
    setb LED_ALBASTRU

system_no_change:
    ret

end
